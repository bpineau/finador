# Finador phase E - v0.2 : plan d'implémentation

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** La v0.2 demandée par l'utilisateur : références courtes non ambiguës (`add cw8`), exclusions d'actifs dans toute portée (`perf --exclude btc,vizr`), ventilation par enveloppe (`value --by enveloppe`, aussi au web), hypothèses jetables (`value --what-if vizr=280` avec delta), `asset edit/rm` (gestion des alias, ISIN, retenue à la source sur dividendes), couleurs ANSI dans `perf`, et verrouillage optimiste inter-processus (CLI ↔ serve).

**Architecture:** tout s'appuie sur l'existant - la résolution gagne un tier « préfixe unique » dans `domain.resolve` ; `Scope` gagne un set d'exclusions ; `portfolio.Value` gagne des options fonctionnelles (ventilation, surcharges de prix) ; `Asset` gagne `Withholding` ; `store.File` gagne l'horodatage optimiste + flock. Conventions inchangées (TDD, gofmt/vet silencieux, français, pas de binaire committé, tests sans réseau).

**Décisions actées phase E :**
- **Références courtes** : un tier de résolution PRÉFIXE s'applique seulement si tous les tiers exacts ont échoué ; il matche le début (insensible à la casse) de l'ID, du ticker, de l'ISIN, d'un alias ou du nom ; plusieurs candidats → `ErrAmbiguous` qui LISTE les candidats. Vaut pour les actifs ET les comptes.
- **What-if** : prix exprimés dans la devise de cotation de l'actif (`vizr=280` = 280 USD si VIZR cote en USD), jamais persistés, uniquement sur `value`. L'affichage donne la valeur hypothétique ET le delta vs réel.
- **Retenue à la source** : champ optionnel `Withholding` (taux décimal) sur l'Asset, appliqué aux dividendes AUTOMATIQUES (net = brut × (1−w)) dans Value() et Series(). Les dividendes manuels sont réputés saisis nets. Édité par `asset edit X --withholding 15%`.
- **Verrou inter-processus** : verrouillage OPTIMISTE - `store.File` retient (taille, mtime ns) du fichier à l'ouverture ; `Save()` prend un flock sur `<db>.lock`, vérifie que le disque n'a pas changé depuis l'ouverture, écrit, met à jour l'horodatage. Conflit → erreur sentinelle `store.ErrConcurrent` (« modifié par un autre processus - relancez »). Un flock DUR bloquerait la CLI pendant `serve` : refusé. Flock unix par build tag, no-op ailleurs (le check d'horodatage reste).
- **Couleurs ANSI** : cellules TWR/XIRR de `perf` en vert/rouge selon le signe ; actives seulement si stdout est un terminal ET `NO_COLOR` absent ; `--no-color` pour forcer ; en test, `FINADOR_FORCE_COLOR=1` permet d'asserter les séquences.
- **`asset rm`** : refuse si une transaction référence l'actif (message expliquant `tx rm`) - pas de --force en v0.2.

```go
// resolve returns the single item matching ref on the first tier that yields
// any match; several matches on the same tier is an ambiguity. When every
// exact tier fails, a unique case-insensitive PREFIX of any reference wins -
// « add cw8 » without remembering the full id.
func resolve[T any](ref, what string, items []*T, tiers ...func(*T) []string) (*T, error) {
	if ref == "" {
		return nil, fmt.Errorf("%s (référence vide): %w", what, ErrNotFound)
	}
	for _, tier := range tiers {
		matches := lo.Filter(items, func(it *T, _ int) bool {
			return lo.SomeBy(tier(it), func(s string) bool {
				return s != "" && strings.EqualFold(s, ref)
			})
		})
		switch len(matches) {
		case 0: // tier suivant
		case 1:
			return matches[0], nil
		default:
			return nil, fmt.Errorf("%s %q: %w", what, ref, ErrAmbiguous)
		}
	}
	// tier préfixe : sur TOUTES les références confondues
	low := strings.ToLower(ref)
	var hits []*T
	var hitIDs []string
	for _, it := range items {
		for _, tier := range tiers {
			found := lo.SomeBy(tier(it), func(s string) bool {
				return s != "" && strings.HasPrefix(strings.ToLower(s), low)
			})
			if found {
				hits = append(hits, it)
				hitIDs = append(hitIDs, tiers[0](it)[0]) // l'ID, premier tier
				break
			}
		}
	}
	switch len(hits) {
	case 1:
		return hits[0], nil
	case 0:
		return nil, fmt.Errorf("%s %q: %w", what, ref, ErrNotFound)
	default:
		return nil, fmt.Errorf("%s %q (candidats : %s): %w",
			what, ref, strings.Join(hitIDs, ", "), ErrAmbiguous)
	}
}
```

```go
func (b *Book) CheckAssetRefs(a *Asset) error {
	refs := append([]string{string(a.ID), a.Ticker, a.ISIN, a.Name}, a.Aliases...)
	for _, other := range b.Assets {
		if other.ID == a.ID {
			continue
		}

	}
	return nil
```

```go
// parseExclusions resolves a comma-or-repeated --exclude list into asset IDs.
func parseExclusions(b *domain.Book, refs []string) (map[domain.AssetID]bool, error) {
	if len(refs) == 0 {
		return nil, nil
	}
	out := map[domain.AssetID]bool{}
	for _, chunk := range refs {
		for _, ref := range strings.Split(chunk, ",") {
			ref = strings.TrimSpace(ref)
			if ref == "" {
				continue
			}
			asset, err := b.Asset(ref)
			if err != nil {
				return nil, fmt.Errorf("--exclude %s: %w", ref, err)
			}
			out[asset.ID] = true
		}
	}
	return out, nil
}
```

```go
// parseWhatIf reads "ref=prix" pairs into asset-ID overrides.
func parseWhatIf(b *domain.Book, pairs []string) (map[domain.AssetID]float64, error) {
	if len(pairs) == 0 {
		return nil, nil
	}
	out := map[domain.AssetID]float64{}
	for _, p := range pairs {
		ref, val, ok := strings.Cut(p, "=")
		if !ok {
			return nil, fmt.Errorf("--what-if %q: attendu actif=prix", p)
		}
		asset, err := b.Asset(strings.TrimSpace(ref))
		if err != nil {
			return nil, fmt.Errorf("--what-if %s: %w", ref, err)
		}
		price, err := strconv.ParseFloat(strings.TrimSpace(val), 64)
		if err != nil || price < 0 {
			return nil, fmt.Errorf("--what-if %s: prix %q invalide", ref, val)
		}
		out[asset.ID] = price
	}
	return out, nil
}

// printWhatIfDelta compares the hypothesis with reality.
func printWhatIfDelta(cmd *cobra.Command, hyp, base portfolio.Valuation) {
```

```go
func ParsePercent(s string) (float64, error) {
	v, err := strconv.ParseFloat(strings.TrimSuffix(strings.TrimSpace(s), "%"), 64)
	if err != nil || v < 0 || v > 100 {
		return 0, fmt.Errorf("pourcentage %q invalide (attendu 0%% à 100%%)", s)
	}
	return v / 100, nil
}
```

```go
}
```

```go
//go:build unix

package store

import (
	"os"
	"syscall"
)

// lockSidecar serializes the short critical section of Save across processes.
func lockSidecar(path string) (func(), error) {
	lf, err := os.OpenFile(path, os.O_CREATE|os.O_RDWR, 0o600)
	if err != nil {
		return nil, err
	}
	if err := syscall.Flock(int(lf.Fd()), syscall.LOCK_EX); err != nil {
		lf.Close()
		return nil, err
	}
	return func() {
		syscall.Flock(int(lf.Fd()), syscall.LOCK_UN)
		lf.Close()
	}, nil
}
```
